
```
! Создание файла с конкретной датой во времени обращения
! 00:00 01.05
touch -t 05010000 f1.png

! Удаление файлов, к которым обращались больше 100 дней назад
find . -atime +100 -delete

! удаление всех файлов к которым не было обращений за последние 100 дней, выводя запрос удаления
find . -atime +100 -exec rm -i {} \;

! использование ; приведет к выполнению нескольких команд (отдельно для каждого аргумента)
! использование + приведет к выполнению наименьшего количества возможных команд

!копировать в указанный каталог все файлы в формате «png», содержащиеся в домашнем каталоге пользователя
find . -name "*.png" -exec cp {} ./new \;

! команда, которая будет перемещать в указанный каталог все файлы в формате «png», содержащиеся в домашнем каталоге пользователя.
find . -maxdepth 1 -name "*.png" -exec mv -v {} ./new \;


! команда, которая будет искать в домашнем каталоге пользователя файлы содержащие фразу «hello» и выводить путь к этим файлам.
find . -type f -exec grep -l 'hello' {} \;
или
find . -type f -exec grep -q 'hello' {} \; -print
```
