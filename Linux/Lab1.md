
```
! Создание файла с конкретной датой во времени обращения
! 00:00 01.05
touch -t 05010000

! Удаление файлов, к которым обращались больше 100 дней назад
find . -atime +100 -delete

! удаление всех файлов к которым не было обращений за последние 100 дней, выводя запрос удаления

find . -atime +100 -exec rm -i {} \;

! использование ; приведет к выполнению нескольких команд (отдельно для каждого аргумента)
! использование + приведет к выполнению наименьшего количества возможных команд
```
